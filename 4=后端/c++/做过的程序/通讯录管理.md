#include <iostream>
using namespace std;
#include <string>
#define MAX  1000

struct Person {
    string m_Name;
    int m_Sex;
    int m_Age;
    string m_Phone;
    string m_Addr;
};

// 通讯录结构

struct addressbooks {
    // 联系人数组
    struct Person personArray[MAX];
    int m_Size;
};


// 菜单界面
void showMenu() {
    cout << "***** 1. 添加联系人 ***** " << endl;
    cout << "***** 2. 显示联系人 ***** " << endl;
    cout << "***** 3. 删除联系人 ***** " << endl;
    cout << "***** 4. 查找联系人 ***** " << endl;
    cout << "***** 5. 修改联系人 ***** " << endl;
    cout << "***** 6. 清空联系人 ***** " << endl;
    cout << "***** 0 退出联系人  ***** " << endl;
};

// 添加用户
void addPerson(addressbooks *abs) {
    // 判断通讯录是否已经满了，满了就不再添加
    if (abs -> m_Size == MAX) {
        cout << "通讯录已经满了" << endl;
        return;
    } else {
        // 添加
        string name;
        cout << "请输入姓名" << endl;
        cin >> name;
        abs -> personArray[abs->m_Size].m_Name = name;

        // 性别sex
        int sex = 0;
        cout << "请输入姓名: " << endl;
        cout << "1 --- 男: " << endl;
        cout << "2 --- 女: " << endl;
        while (true) {
            // 如果输入的是 1 或者 2 就使用，否则就不使用
            cin  >> sex;
            if (sex == 1 || sex == 2) {
                abs->personArray[abs->m_Size].m_Sex = sex;
                break;
            };
            cout << "输入的数据有误，重新输入" << endl;
        }

        // 年龄

        cout << "请输入年龄: " << endl;
        int age = 0;
        cin >> age;
         abs->personArray[abs->m_Size].m_Age = age;
        // 电话
        cout << "请输入电话: " << endl;
        string phone ;
        cin >> phone;
         abs->personArray[abs->m_Size].m_Phone = phone;
        // 住址
        cout << "请输入住址: " << endl;
        string address ;
        cin >> address;
        abs->personArray[abs->m_Size].m_Addr = address;

        // 人数更新
        abs->m_Size++;

        // 提示
        cout << "添加成功" <<endl;
        system("pause");
        system("cls");
    }
}


// 显示所有的联系人
void showPerson(addressbooks *abs) {
    // 通讯录为空 ，提示为空，不空，则显示记录人信息
    if (abs->m_Size == 0) {
        cout << "当前记录为空" << endl;
    } else {
        for (int i=0; i<abs->m_Size; i++) {
            cout << "姓名：" << abs->personArray[i].m_Name << "\t";
            cout << "性别：" << abs->personArray[i].m_Sex << "\t";
            cout << "年龄：" << abs->personArray[i].m_Age << "\t";
            cout << "电话：" << abs->personArray[i].m_Phone << "\t";
            cout << "地址：" << abs->personArray[i].m_Addr << endl;
        };
        system("pause");
        system("cls");
    }
}


// 检查联系人是否存在，存在返回具体的位置，不存在返回 -1
int isExist(addressbooks * abs, string name) {
    for (int i=0; i< abs->m_Size; i++) {
        if (abs->personArray[i].m_Name == name) {
            // 找到返回其下标
            return i;
        };
    };
    return -1;
}


// 删除指定的联系人
void deletePerson(addressbooks *abs) {
    cout << "请输入您要删除的联系人" << endl;
    string name;
    cin >> name;
    int res = isExist(abs,name);
    if (res == -1) {
        cout << "查无此人" << endl;
    } else {
        if (res != abs->m_Size){
            // 数据移动,最后一个直接复制过来
            abs->personArray[res] = abs->personArray[abs->m_Size];
        };
        // 转换直接删除最后一个元素
        abs->m_Size--;
        cout << "删除成功" << endl;
    };
    system("pause");
    system("cls");
}



// 查询指定的人名
void findperson(addressbooks *abs) {
    cout << "请输入要删除的名字" << endl;
    string name;
    cin >> name;

    int res = isExist(abs,name);
    if (res == -1) {
        cout << "查无此人" << endl;
    } else {
        cout << "找到此人" << endl;
    }
}

// 修改指定联系人信息
void modifyPerson(addressbooks * abs) {
    cout << "请输入要修改的联系人:" << "\t";
    string name;
    cin >> name;
    int res = isExist(abs,name);
    if (res == -1) {
        cout << "查无此人" << endl;
    } else {
        // 找到该人
        // 姓名
        cout << "请输入姓名： ";
        string name;
        cin >> name;
        cout << endl;
        abs->personArray[res].m_Name = name;
        // 性别

        while (true) {
            cout << "请输入性别： ";
            int  sex;
            cin >> sex ;
            cout << endl;
            if (sex == 1 || sex == 2) {
                abs->personArray[res].m_Sex = sex;
                break;
            }
            cout << "输入错误，重新输入" << endl;
        }

        // 年龄
        cout << "请输入年龄： ";
        int age;
        cin >> age;
        cout << endl;
        abs->personArray[res].m_Age = age;
        // 电话
        cout << "请输入电话： ";
        string phone;
        cin >> phone;
        abs->personArray[res].m_Phone = phone;
        cout << endl;
        // 地址
        cout << "请输入家庭地址： ";
        string addr;
        cin >> addr;
        abs->personArray[res].m_Addr = addr;
        cout << endl;
    };
    system("pause");
    system("cls");
}


// 清除信息
void cleanPerson(addressbooks * abs) {
    while (true) {
        cout << "请确认是否要全部清空,确定请安1 ，否则请按 2 " ;
        int key;
        cin >> key;
        switch (key) {
        case 1 :
            abs->m_Size = 0;
            cout << " 已经全部清空" << endl;
            return;
        case 2 :
            cout << "取消操作" << endl;
            return;
        default :
            cout << "输入错误，请重新输入 " << endl;
        }

    }


}


int main() {

    // 创建通讯录结构体变量
    struct addressbooks abs;
    abs.m_Size = 0;

    // 用户选择输入的变量
    int select = 0;
    while (true) {
        showMenu();
        cin >> select;
        switch (select) {
        case 1: // 添加
            addPerson(&abs);
            break;
        case 2: //
            showPerson(&abs);
            break;
        case 3: // 删除
            deletePerson(&abs);
            break;
        case 4:  // 查找
            findperson(&abs);
            break;
        case 5:  // 修改
            modifyPerson(&abs);
            break;
        case 6:  // 清除
            cleanPerson(&abs);
            break;
        case 0:
            cout << "欢迎下一次使用" << endl;
            break;
        default:
            break;
        }
    }

    return 0;
}
