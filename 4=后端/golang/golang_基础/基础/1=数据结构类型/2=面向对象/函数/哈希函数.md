func main() {
    TestString := "http://c.biancheng.net/golang/"
    // return   hash.Hash
    Md5Inst := md5.New()
    
    Md5Inst.Write([]byte(TestString))
    
    Result := Md5Inst.Sum([]byte(""))
    
    fmt.Printf("%x\n\n", Result) // 6dc42d81095839903edf352ef1ec0a6a
    
    Sha1Inst := sha1.New()
    Sha1Inst.Write([]byte(TestString))
    Result = Sha1Inst.Sum([]byte(""))
    fmt.Printf("%x\n\n", Result)  // 32313d69e3f0e4bbf6738858274e7e2c9a46d293
}


// 对文件内容 计算 sha1 
func main() {
    TestFile := "123.txt"
    infile, inerr := os.Open(TestFile)
    if inerr == nil {
        md5h := md5.New()
        io.Copy(md5h, infile)
        fmt.Printf("%x %s\n", md5h.Sum([]byte("")), TestFile)  // 6dc42d81095839903edf352ef1ec0a6a 123.txt
        sha1h := sha1.New()
        io.Copy(sha1h, infile)
        fmt.Printf("%x %s\n", sha1h.Sum([]byte("")), TestFile) // da39a3ee5e6b4b0d3255bfef95601890afd80709 123.txt
    } else {
        fmt.Println(inerr)
        os.Exit(1)
    }
}
