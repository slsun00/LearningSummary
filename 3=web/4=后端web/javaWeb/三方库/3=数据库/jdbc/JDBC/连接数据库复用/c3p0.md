## src/utils/JDBCUtils

```java
// 重复代码抽取

public class JDBCUtils {

    
public static Connection getConnection(){
	// 什么都不传，使用的是默认的连接池
    // CombopooledDataSource("") , 使用 named-config name="..."
        return new CombopooledDataSource().getConnection();
}
public static void close(Statement stmt, Connection conn){
    if (stmt != null) {
        try {
            stmt.close();
        } catch(SQLEXcption e){
            e.printStackTrace();
        }
    }

    if (conn != null) {
        try {
            stmt.close();
        } catch(SQLEXcption e){
            e.printStackTrace();
        }
    }
}
public static void close(ResuktSet res, Statement stmt, Connection conn){
		if (stmt != null) {
            try {
                stmt.close();
            } catch(SQLEXcption e){
                e.printStackTrace();
            }
        }
        
        if (conn != null) {
            try {
                stmt.close();
            } catch(SQLEXcption e){
                e.printStackTrace();
            }
        }
        
        if (res != null) {
            try {
                stmt.close();
            } catch(SQLEXcption e){
                e.printStackTrace();
            }
        }
    }
}
```
