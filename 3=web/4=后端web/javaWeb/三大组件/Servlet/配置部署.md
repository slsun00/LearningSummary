## 默认

```java
介绍
	Servlet 应用程序位于路径 <Tomcat-installation-directory>/webapps/ROOT 下
	类文件放在 <Tomcat-installation-directory>/webapps/ROOT/WEB-INF/classes 中。
    
例子
    // 完全合格的类名称 com.myorg.MyServlet
     Servlet 类必须位于 WEB-INF/classes/com/myorg/MyServlet.class 中，
         
注意
	IDEA 部署， 不需要讲究这么多         
```



## web.xml

```xml
<?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns="http://xmlns.jcp.org/xml/ns/javaee"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://xmlns.jcp.org/xml/ns/javaee http://xmlns.jcp.org/xml/ns/javaee/web-app_4_0.xsd"
         version="4.0">
	<!--tomcat 匹配值 Servlet 程序-->
    <servlet>
		<!--servlet 程序起一个别名，一般是类名-->
        <servlet-name>HelloServet</servlet-name>
		<!--servlet 程序全类名。类地址-->
        <servlet-class>Servlet.HelloServlet</servlet-class>
    </servlet>

	<!--给servlet 程序配置访问地址-->
    <servlet-mapping>
		<!--告诉服务器，当前配置的地址给哪个 servlet 程序使用，类名-->
        <servlet-name>HelloServet</servlet-name>
		<!-- 配置访问地址  / 表示服务器在解析的时候，表示为： http://ip:port/工程路径-->
		<!-- /hello   http://ip:port/工程路径/hello-->
        <url-pattern>/hello</url-pattern>
    </servlet-mapping>

</web-app>
```
