### 表格全选

```js
介绍
	让所有复选框的 checked 属性（选中状态），跟随全选按钮即可
语法案例
		<script type="text/javascript">
			var j_cdAll = document.getElementById('j_cdAll')
			var j_tbs = document.getElementById('j_tb').getElementsByTagName('input')
			// 点击全选，选择全部
			j_cdAll.onclick = function(){
				if (this.checked){
					for (var i=0; i<j_tbs.length; i++){
						j_tbs[i].checked = this.checked;
					}
				}
			}
			// 点击所有小的，锁定全部
			for (var i=0; i<j_tbs.length; i++){
				j_tbs[i].onclick = function(){
					// flag 记录是否全部选中
					var flag = true;
					// 每次点击小按钮，就要循环遍历所有的小按钮，看是否都选中了
					for  (var i=0; i<j_tbs.length; i++){
						if (!j_tbs[i].checked){
							flag = false;
                               break;
						}
					}
					j_cdAll.checked = flag;
				}
			}
		</script>
	</body>
```

### 动态表格

```js
		<script type="text/javascript">
			var tbody = document.querySelector('tbody')
			for (var i=0; i<datas.length; i++){
				// 创建行

						// 父（字）
						tbody.removeChild(this.parentNode.parentNode);
					}
				}
				
				
			}
		</script>
```

## 评论增加/删除

```js
介绍
	点击按钮以后，创建一个 li , 添加到 ul 中， 同时将文本域中的值通过 html.innerHTML 赋值给 li
    后续增减，进行 childrenAppend
语法例子
		<script type="text/javascript">
			var btn = document.querySelector('button')
			var text = document.querySelector('textarea')
			var ul = document.querySelector('ul')
			btn.onclick = function(){
				if (text.value == ''){
					alert("没有输入内容");
					return false;
				} else {
					// 创建、添加
					var li = document.createElement('li');
					li.innerHTML = text.value + '<a href='#'>删除</a>';
					// ul.appendChild(li);				
					ul.insertBefore(li,ul.children[0])
                    
                    // 删除
                    var as = document.querySelectorAll('a')
                    for (var i=0; i<as.length; i++){
                        as[i].onclick = function(){
                            // this.parentNode
                            ul.removeChild(this.parentNode)
                        }
                    }
				}				
			}
		</script>
```

## 鼠标动态图

```js
原理
	鼠标每次移动，都会获得新的鼠标坐标，把这个 x 和 y 坐标作为图片的 top 和 left 值
		<script type="text/javascript">
			
			var pic = document.querySelector('img')
			document.addEventListener('mousemove',function(e){
				var x = e.pageX;
				var y = e.pageY;
				// 走图片的一半， 让鼠标集中到图片的中间
				pic.style.left = x+'px';
				pic.style.top = y + 'px';				
			})
		</script>
```

## 输入框字体上行字体变大

```js
<script type="text/javascript">
    var con = document.querySelector('.con')
    var jd_input = document.querySelector('.jd')
    jd_input.addEventListener('keyup', function(){
        if (this.value === ''){
            con.style.display = 'none'
        } else {
            con.style.display = 'block'
            con.innerText = this.value;
        }
    })
	// 失去焦点， 隐藏盒子
    jd_input.addEventListener('blur', function(){
        if (this.value === ''){
            con.style.display = 'none'
        }
    })
	// 获得焦点， 就显示盒子
    jd_input.addEventListener('focus', function(){
        if (this.value !== ''){
            con.style.display = 'block';
        } 
    })
</script>
```
