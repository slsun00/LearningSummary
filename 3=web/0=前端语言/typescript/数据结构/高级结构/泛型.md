### 泛型函数

```ts
1. 要求传入的参数和返回的参数一致
	//  第一个 T 就是在调用过的时候约定传入参数的类型
	function getData<T>(value:<T>) :T {
        return value
        // 加入你传入的是 number 类型，但是 return '234' 是不可以的
    }
	// 调用
	// 外部调用的时候，要求传入的参数是 number 类型     
	getData<number>(123)

2. 返回参数和传入参数不一致
	function getData(value:<T>) :any { }
```



### 泛型类

```ts
class Minclass<T> {
    public list:T[] = []
    add(value:T) :void {
        this.list.push(value)
    }
}
// T 为 number
var num1 = new Minclass<number>()
num1.add(1)


// 类作为参数的泛型类
class User {
    userName:string | undefined
    pwd:string | undefined
}
class Mysql<T> {
    add(user:T):boolean {        
    }
}

// 这个比接口多的就是可以实例化
var a = new User()
// 对数据进行校验
var db = new Mysql<User>()
db.add(a)
```

### 泛型接口

```ts
方式一
	interface Config {
        <T>(value:T) :T
    }
    // 实现 
    var getData:Config = function<T>(value:T) :T { ... }
    // 使用
    getData<string>('张三')
                                                  
方式二
	interface Config<T> {
        (value:T) :T
    }
    function getData<T>(value:T) :T { ... }
    // 调用
    var myget:Config<string> = getData
    myget("sa")
```
