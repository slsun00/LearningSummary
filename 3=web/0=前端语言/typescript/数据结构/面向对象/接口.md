### 属性接口(对象约束)

```ts
介绍
	对 json 进行约束（参数是对象字面量）
 例子
 	interface FullName {
        firstName:string
        secondName:string
        age?:string
    }
	// 接口对方法进行约束
	function printName(name:Fullname) { ... }
    function printName(name:{firstName:string,secondName:string}) { ... }
    // 调用
    // 对象实现接口，键值对形式固定传入
    let obj = {firstName:'li',secondName:'li'}
    // 对象传入函数
    printName(obj)
```

### 函数类型接口

```ts
介绍
	对传入的参数以及返回值进行约束
 例子
 	interface encrypt {
        // (函数参数)返回值
        (key:string,value:string):string
    }
    // 接口实现
    var md5:encrypt = function(key:string,value:string) :string {}
    // 调用
    md5('1','值')
```

### 可索引接口(数组)

```ts
介绍
	数组、对象的约束(不常用)、
例子
	interface User {
        [index:number] :string
    }
	// 实现
	var a:User = [123,234]
    ------------------------------
    interface User {
        // 下表不是数字，就是对象了
        [index:string] :string
    }
	var a:User = {name:'23'}
```

### 类类型接口(类)

```ts
介绍
	对类进行约束，和抽象类相似
例子
	interface Animal {
        name:string
        eat(str:string) :void
    }
    
    // 类实现接口,必须和父类完全，
    class Dog implements Animal {
        name:string
        constructor(name:string) {
            this.name = name
        }
        // 不能有参数
        eat() {
            console.log('吃粮食')
        }
    }
	// 调用
	var dog = new Dog('小黑')
    dog.eat()
```



## 接口继承

```ts
interface Animal {
    eat() :void
}
// 继承
interface Person extends Animal {
    work() :void
}
// 实现
class Web implements Person {
    public name:string
    constructor(name:string){
        this.name = name
    }
    eat() {}
    work() {}
    sing() {}
}
// 调用
let a = new Web()
a.work
-----------------------
// 继承 + 实现
class Programmer {
    public name:string
    constructor(name:string){
        this.name = name
    }
    coding() {}
}
class Web extends Programmer implements Person {
    public name:string
    constructor(name:string){
        this.name = name
    }
    eat() {}
    work() {}
    sing() {}
} 
// 调用
let a = new Web()
a.work
```
