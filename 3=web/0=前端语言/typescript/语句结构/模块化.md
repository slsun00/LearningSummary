## 内部空间

```ts
namespace A {
    class a {
        constructor(){}
        name = 'xiaoxiao'
    }
}

namespace B {
    constructor(){}
    class a {
        name = 'xiaoxiao'
    }
}
// 上面代码在一个文件中，是可以存在的 , 不存在命名冲突
// 调用
    let a = new A.a()
    
//可以在命名空间内部导出
namespace A {
    export class a {
        constructor(){}
        name = 'xiaoxiao'
    }
}
```
