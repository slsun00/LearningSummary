## 二叉树

```go
// 孩子表示法
type TreeNode struct {
    Data string      //节点用来存放数据
    Left *TreeNode   // 左子树
    Right *TreeNode  // 右子树
}
```

### 前序遍历

```go
// 访问顺序  ： 根结点 -- 左子树 --- 右子树
func PreOrder(node *TreeNode) {
	
    if node == nil {      			// 这个是当前节点是否为空
        return
    }
    fmt.Print(node.Data,"")			// 非空打印当前元素的data
    PreOrder(node.Left)				// 打印左子树
    PreOrder(node.Right)			// 打印右子树
    
}
```

### 中序遍历

```go
func MidOrder(node *TreeNode) {
    
    if node == nil { 			//当前节点是否为空
        return
    }
    MidOrder(node.Left)
    MidOrder(node.Right)
    fmt.Print(node.Data,"")
}
```



### 后序遍历

```go
func PostOrder(node *TreeNode) {
    if node == nil {
        return
    }
    PostOrder(node.Left)
    PostOrder(node.Right)
    fmt.Print(node.Data,"")
}
```
