package main

import "fmt"

const maxSize int = 20

type data interface{}

type Stack struct {
	element []data
	len     int
}

// 创建一个新栈
func CreateStack() *Stack {
	return &Stack{
		len:     0,
		element: make([]data, 0, maxSize),
	}
}

// 判断是否为空
func (s *Stack) IsEmpty() bool {
	if s.len == 0 {
		return true
	}
	return false
}

// 求栈的长度
func (s *Stack) Length() int {
	return s.len
}

// 进栈
func (s *Stack) Push(data data) {
	s.element = append(s.element, data)
	s.len++
}

// 查询栈顶元素
func (s *Stack) TopElement() data {
	if ok := s.IsEmpty(); ok {
		return nil
	}
	return s.element[s.len-1]
}

// 出栈操作
func (s *Stack) Pop() {
	if ok := s.IsEmpty(); ok {
		fmt.Println("空栈无法弹出")
		return
	}
	s.element = s.element[:s.len-1]
	s.len--
}

// 打印栈
func (s *Stack) Travel() {

	// 直接打印
	// fmt.Println(s.element)
	top := -1
	for {
		if top == s.len-1 {
			return
		}
		top++
		fmt.Printf("%5v", s.element[top])
	}
}

func main() {
	s := CreateStack()
	s.Push(2)
	s.Push(4)
	s.Push(6)
	s.IsEmpty()
	s.Travel()
	fmt.Println("---------------")
	s.Pop()
	s.Travel()
}
