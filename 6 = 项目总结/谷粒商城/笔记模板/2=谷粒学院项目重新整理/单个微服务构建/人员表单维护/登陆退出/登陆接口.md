### 前端

#### 登陆接口

src/api/login.js，可以在这个文件中分析后端Mock数据的接口的url地址：
登录：/user/login
登出：/user/info
获取用户信息：/user/logout

### 后端

```java
package com.atguigu.guli.service.edu.controller;
@CrossOrigin //跨域
@RestController
@RequestMapping("/user")
public class LoginController {

    /**
     * 登录
     * @return
     */
    @PostMapping("login")
    public R login() {
        return R.ok().data("token","admin");   // 前端有一个拦截器
    }

    /**
     * 获取用户信息
     * @return
     */
    @GetMapping("info")
    public R info() {
        return R.ok()
                .data("roles","[admin]")
                .data("name","admin")
			 .data("avatar","https://oss.aliyuncs.com/aliyun_id_photo_bucket/default_handsome.jpg");
    }

    /**
     * 退出
     * @return
     */
    @PostMapping("logout")
    public R logout(){
        return R.ok();
    }
}
```
